Creating new Git repository with the GitHub CLI 

        
        #-------------------------------------------------------------+
        #                 creating Git repository
        #-------------------------------------------------------------+
        function create_private_git_repo(){
           echo "+------------------------------------+"
           echo "|         Creating Repository        |"
           echo "+------------------------------------+"
           read -p "Create repo? [Y/n]: " DECISION
           echo "Decision: $DECISION"
           if [ $DECISION = 'Y' ];
           then
             read -p "Enter new repo name (prefix): " REPO_NAME
             gh repo create $REPO_NAME --private
             echo "*** Repository '$REPO_NAME' created  ***"
             sleep 0.5
        
             echo "*** Cloning new repo ***"
             read -p "Enter GitHub username: " GITHUB_USERNAME
             git clone "https://github.com/$GITHUB_USERNAME/$REPO_NAME.git"
        
             echo "*** Changing to directory $REPO_NAME ***"
             cd $REPO_NAME
           else
             echo "*** Repo NOT created ***"
           fi
        }
