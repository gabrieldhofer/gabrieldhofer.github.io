function IsComment()
  if getline('.') =~ '^\s*//'
    return 1
  else
    return 0
  endif
endfunction

function ToggleComment()
        let is_commented = IsComment()
        if ! is_commented
                normal ^i//
                normal j
        else
                normal ^xxj
        endif
endfunction
