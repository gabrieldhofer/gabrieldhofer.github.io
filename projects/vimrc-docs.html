function IsComment()
  if getline('.') =~ '^\s*//'
    return 1
  else
    return 0
  endif
endfunction

function ToggleComment()
        let is_commented = IsComment()
        if ! is_commented
                normal ^i//
                normal j
        else
                normal ^xxj
        endif
endfunction

map <space> :call ToggleComment()<CR>
Expanding/Collapsing For Loops & removing whitespace between paraentheses
function MyWhitespaceFormatting()
        :s/(\s*/(/ge
        :s/\s*)/)/ge
endfunction

map q <Nop>
map qf [(v])"
map qh :call MyWhitespaceFormatting()<CR>

function MakeOneLine()
        normal [(v])"
        :call MyWhitespaceFormatting()
endfunction

map qg :call MakeOneLine()<CR>
